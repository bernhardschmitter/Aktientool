name: Update Prices

on:
  workflow_dispatch:
  schedule:
    # Montag-Freitag abends nach Börsenschluss, deutsche Zeit je nach Sommer/Winter leicht versetzt
    - cron: '30 20 * * 1-5'

permissions:
  contents: write

jobs:
  update-prices:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'

      - name: Generate prices.json from Stooq EOD data
        run: node .github/scripts/update-prices.js

      - name: Commit updated prices.json
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "41898282+github-actions[bot]@users.noreply.github.com"
          git add prices.json
          if git diff --cached --quiet; then
            echo "No price changes to commit"
          else
            git commit -m "Update EOD prices"
            git push
          fi
